  for word in re.split(r'([ |\n|=|.|,|(|)|[|]|<|>|"|:|;|-|/|–|%])',new_string):
    if type(word)==str:
      if word in delims:
        string_to_print += word
      elif word in count_to_word_dict.keys():
        if len(count_to_word_dict[word])>0:
          string_to_print += count_to_word_dict[word][0]+' '
      else:
        pass
